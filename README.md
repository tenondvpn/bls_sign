# Powered by libBLS: a C++ library for BLS Threshold Signatures

## build
git clone git@github.com:tenondvpn/bls_sign.git

cd ./bls_sign/deps && sh build.sh

cd ../ && mkdir -p build && cd build && cmake .. && make -j4
